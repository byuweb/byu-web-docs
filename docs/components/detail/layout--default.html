$emphasisMargin: ( ($emphasisPercent - 100) / 2 ) * -1
$featurePercent: ( $featureWidth / $columnWidth ) * 100%
$featureMargin: ( ($featurePercent - 100) / 2 ) * -1

	margin-left: $emphasisMargin
